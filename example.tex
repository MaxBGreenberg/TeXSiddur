\section{והשיאנו}

\section{הקדוש ברוך הוא}
הַקָּדוֹשׁ בָּרוּךְ הוּא, אֲנַחְנוּ אוֹהֲבִים אוֹתָךְ.
\section{והאר עינינו}
וְהָאֵר עֵינֵֽינוּ בְּתוֹרָתֶֽךָ, וְדַבֵּק לִבֵּֽנוּ בְּמִצְוֹתֶֽיךָ, וְיַחֵד לְבָבֵֽנוּ לְאַהֲבָה וּלְיִרְאָה אֶת־שְׁמֶֽךָ֭, וְלֹא נֵבוֹשׁ לְעוֹלָם וָעֶד.
\section{סוכת דוד}
הָרַחֲמָן הוּא יָקִים לָֽנוּ אֶת־סֻכַּת דָּוִד הַנּוֹפָֽלֶת.
\section{אבינו מלכנו}
אָבִֽינוּ מַלְכֵּֽנוּ חָנֵּֽנוּ וַעֲנֵֽנוּ כִּי אֵין בָּֽנוּ מַעֲשִׂים עֲשֵׂה עִמָּֽנוּ צְדָקָה וָחֶֽסֶד וְהוֹשִׁיעֵֽנוּ.
\section{והיא שעמדה}
וְהִיא שֶׁעָמְדָה לַאֲבוֹתֵֽינוּ וְלָֽנוּ. שֶׁלֹּא אֶחָד בִּלְבָד עָמַד עָלֵֽינוּ לְכַלּוֹתֵֽנוּ, אֶלָּא שֶׁבְּכׇל־דּוֹר וָדוֹר עוֹמְדִים עָלֵֽינוּ לְכַלוֹתֵֽנוּ, וְהַקָּדוֹשׁ בָּרוּךְ הוּא מַצִּילֵֽנוּ מִיָּדָם.
\section{למענך}
לְמַעַנְךָ ﭏֱֺהֵֽינוּ עֲשֵׂה וְלֹא לָֽנוּ. רְאֵה עֲמִידָתֵֽנוּ דַּלִּים וְרֵקִים. הַנְּשָׁמָה לָךְ, וְהַגּוּף פָּעֳלָךָ חוּסָה עַל עֲמָלָךְ.
\section{חמול}
חֲמוֹל עַל מַעֲשֶֽׂיךָ, וְתִשְׂמַח בְּמַעֲשֶֽׂיךָ, וְיֹאמְרוּ לְךָ חוֹסֶֽיךָ בְּצַדֶּֽקְךָ עֲמוּסֶֽיךָ. תֻּקְדַּשׁ אָדוֹן עַל כׇּל־מַעֲשֶׂיךָ, כִּי מַקְדִּישֶֽׁיךָ בִּקְדֻשָּׁתְךָ קִדַּשְׁתָּ. נָאֶה לְקָדוֹשׁ פְּאֵר מִקְּדוֹשִׁים.
\section{שמחה לארצך}
שִׂמְחָה לְאַרְצֶךָ, וְשָׂשׂוֹן לְעִירֶךָ, וּצְמִיחַת קֶרֶן לְדָוִד עַבְדֶּךָ, וַעֲרִיכַת נֵר לְבֶן יִשַׁי מְשִׁיחֶךָ, בִּמְהֵרָה בְיָמֵינוּ.
\section{שמע ישראל}
שְׁמַע יִשְׂרָאֵל ײֳ ﭏֱֺהֵֽינוּ ײֳ אֶחָד.
\section{הוא אלהינו}
הוּא ﭏֱֺהֵֽינוּ הוּא אָבִֽינוּ. הוּא מַלְכֵּֽנוּ הוּא מושִׁיעֵֽנוּ. וְהוּא יַשְׁמִיעֵֽנוּ בְּרַחֲמָיו שֵׁנִית לְעֵינֵי כׇּל־חָי לִהְיות לָכֶם לֵﭏֺהִים. אֲנִי ײֳ ﭏֱֺהֵיכֶם.
\section{אדיר אדירנו}
אַדִּיר אַדִּירֵֽנוּ ײֳ אֲדנֵֽינוּ מָה אַדִּיר שִׁמְךָ בְּכׇל־הָאָֽרֶץ. וְהָיָה ײֳ לְמֶֽלֶךְ עַל כׇּל־הָאָֽרֶץ. בַּיּום הַהוּא יִהְיֶה ײֳ אֶחָד וּשְמו אֶחָד.
\section{ממקומך}
מִמְּקוֹמְךָ מַלְכֵּֽנוּ תוֹפִיעַ וְתִמְלוֹךְ עָלֵֽינוּ כִּי מְחַכִּים אֲנַחְנוּ לָךְ. מָתַי תִּמְלֺךְ בְּצִיּון, בְּקָרוֹב בְּיָמֵֽינוּ לְעוֹלָם וָעֶד תִּשְׁכּוֹן. תִּתְגַּדַּל וְתִתְקַדַּשׁ בְּתוֹךְ יְרוּשָׁלַיִם עִירְךָ לְדוֹר וָדוֹר וּלְנֵֽצַח נְצָחִים. וְעֵינֵֽינוּ תִרְאֶֽינָה מַלְכוּתֶֽךָ כַּדָּבָר הָאָמוּר בְּשִׁירֵי עֻזֶּֽךָ עַל יְדֵי דָּוִד מְשִֽׁיחַ צִדְקֶֽךָ.
\section{אין קצבה}
אֵין קִצְבָה לִשְׁנוֹתֶיֽךָ. וְאֵין קֵץ לְאֹֽרֶךְ יָמֶֽיךָ. וְאֵין לְשַׁעֵר מַרְכְּבוֹת כְּבוֹדֶֽךָ. וְאֵין לְפָרֵשׁ עֵלוּם שְׁמֶֽךָ. שִׁמְךָ נָאֶה לְךָ וְאַתָּה נָאֶה לִשְׁמֶֽךָ. וּשְׁמֵֽנוּ קָרָֽאתָ בִּשְׁמֶֽךָ. עֲשֵׂה לְמַֽעַן שְׁמֶֽךָ וְקַדֵּשׁ אֶת־שִׁמְךָ עַל מַקְדִּישֵׁי שְׁמֶֽךָ. בַּעֲבוּר כְּבוֹד שִׁמְךָ הַנַּעַרָץ וְהַנִּקְדָּשׁ. כְּסוֹד שִֽׂיחַ שַׂרְפֵי קֹדֶשׁ הַמַּקְדִּישִׁים שִׁמְךָ בַּקֹּֽדֶשׁ. דָּרֵי מַֽעְלָה עִם דָּרֵי מַֽטָּה קוֹרְאִים וּמְשַׁלְּשִׁים בְּשִׁלּוּשׁ קְדֻשָּׁה בַּקֹּֽדֶשׁ.
\section{דוד מלך ישראל}
דָּוִד מֶֽלֶךְ יִשְׂרָאֵל חַי וְקַיָּם.
\section{סימן טוב ומזל טוב}
סִימָן טוֹב וּמַזָּל טוֹב יְהֵא לָֽנוּ יְהֵא לָנוּ וּלְכׇל־יִשְׂרָאֵל. אָמֵן.
\section{אשרינו}
אַשְׁרֵֽינוּ מַה־טּוֹב חֶלְקֵֽנוּ וּמַה־נָּעִים גּוֹרָלֵֽנוּ.
\section{אשריכם}
אַשְׁרֵֽיכֶם תַּלְמִידֵי חַכָמִים שֶׁדִּבְרֵי תוֹרָה חֲבִיבִין עֲלֵֽיכֶם בּיוֹתֵר.
\section{אור זרוע}
אוֹר זָרֻֽעַ לַצַּדִּיק וּלְיִשְׁרֵי־לֵב שִׂמְחָה.
\section{ימים על ימי מלך}
יָמִים עַל־יְמֵי־מֶלֶךְ תּוֹסִיף שְׁנוֹתָיו כְּמוֹ־דֹר וָדֹר.
\section{נחמו}
נַחֲמוּ נַחֲמוּ עַמִּי יֹאמַר ﭏֱֺהֵיכֶם. דַּבְּרוּ עַל לֵב יְרוּשָׁלַ‍יִם וְקִרְאוּ אֵלֶיהָ כִּי מָלְאָה צְבָאָהּ כִּי נִרְצָה עֲו‍ֹנָהּ כִּי לָקְחָה מִיַּד ײֳ כִּפְלַיִם בְּכׇל־חַטֹּאתֶיהָ. קוֹל קוֹרֵא בַּמִּדְבָּר פַּנּוּ דֶּרֶךְ ײֳ יַשְּׁרוּ בָּעֲרָבָה מְסִלָּה לֵﭏֺהֵינוּ.
